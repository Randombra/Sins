function SummonJimmy()
	local gui = Instance.new("ScreenGui")
	gui.Name = "LilJimmyWatcher"
	gui.IgnoreGuiInset = true
	gui.DisplayOrder = 9999
	
	local success, coreGui = pcall(function() return game:GetService("CoreGui") end)
	if success then
		gui.Parent = coreGui
	else
		gui.Parent = LocalPlayer:WaitForChild("PlayerGui")
	end

	local img = Instance.new("ImageLabel")
	img.Parent = gui
	img.Name = "Jimmy"
	img.BackgroundTransparency = 1
	img.Size = UDim2.new(0, 0, 0, 0)
	img.AnchorPoint = Vector2.new(0.5, 0.5)
	img.Position = UDim2.new(0.5, 0, 0.5, 0)
	
	local sound = Instance.new("Sound")
	sound.Name = "JimmyLoop"
	sound.SoundId = Jimmy_SoundID
	sound.Volume = 10
	sound.Looped = true
	sound.Parent = SoundService
	sound:Play()
	
	local cc = Instance.new("ColorCorrectionEffect")
	cc.Parent = Lighting
	cc.Contrast = 2
	cc.Saturation = 0.5
	cc.TintColor = Color3.fromRGB(255, 0, 0)
	
	local blur = Instance.new("BlurEffect")
	blur.Parent = Lighting
	blur.Size = 0

	Lighting.Ambient = Color3.fromRGB(255, 0, 0)
	Lighting.OutdoorAmbient = Color3.fromRGB(255, 0, 0)
	Lighting.FogColor = Color3.fromRGB(255, 0, 0)
	Lighting.TimeOfDay = "00:00:00" 
	
	if writefile and readfile and request and getcustomasset then
		pcall(function()
			local response = request({Url = LilJimmy_URL, Method = "GET"})
			if response.Body then
				writefile("LilJimmy_Cache.png", response.Body)
				img.Image = getcustomasset("LilJimmy_Cache.png")
			end
		end)
	end
	
	local tweenInfo = TweenInfo.new(0.5, Enum.EasingStyle.Exponential, Enum.EasingDirection.In)
	
	TweenService:Create(img, tweenInfo, {Size = UDim2.new(2, 0, 2, 0), Rotation = 15}):Play()
	TweenService:Create(blur, tweenInfo, {Size = 20}):Play()
	TweenService:Create(workspace.CurrentCamera, tweenInfo, {FieldOfView = 120}):Play()

	RunService.RenderStepped:Connect(function()
		local shake = math.random(-50, 50) / 10
		local shakeY = math.random(-50, 50) / 10
		workspace.CurrentCamera.CFrame = workspace.CurrentCamera.CFrame * CFrame.new(shake, shakeY, 0)
	end)
end
